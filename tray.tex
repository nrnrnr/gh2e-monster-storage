  \node [image] at (#2) {\includegraphics[#1]{named/#3.png}};

}

\newcommand\monsterx[3][]{%
  \node [image,#1] at (#2) {\includegraphics{named/#3.png}};

}

\newcommand\fullmonster[4]{% coordinate image path imageparms outlineparms
  \node [image,#4] at (#1) {\includegraphics{named/#2.png}};

}

  \node [image,#4] at (#1) {\includegraphics[#5]{named/#2.png}};
  \draw [outline,#4] (#1) \usepath{#3};
}
