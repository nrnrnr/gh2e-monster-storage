\newcommand\monsterbase[3][]{%
  \node [image] at (#2) {\includegraphics[#1]{named/#3.png}};
  \draw [outline] (#2) \usepath{#3#1};
  \draw [red,fill] (#2) \expandafter+\usecentroid{#3#1} circle (1mm);
}

\newcommand\monsterx[3][]{%
  \node [image,#1] at (#2) {\includegraphics{named/#3.png}};
  \draw [outline,#1] (#2) \usepath{#3};
  \node at (0,0) {
      \xdef\drawme{%
    \noexpand\draw
       [blue,fill,rotate around={\unangle[#1]:(#2)}] (#2)
          \noexpand\expandafter+\noexpand\usecentroid{#3} circle (1mm)
      }};
%  \draw [blue,fill] (#2) \expandafter+\usecentroid{#3#1} circle (1mm);
  \drawme;
}
