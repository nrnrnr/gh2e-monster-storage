\newcommand\finger[2][9.5mm]{%
  \draw [thick,fill=white, fill opacity=0.6] #2 circle (#1);

\newcommand\monsterbase[3][]{%
  \node [image] at (#2) {\includegraphics[#1]{named/#3.png}};
  \draw [outline] (#2) \usepath{#3#1};
  \ifdots
    \draw [red,fill] (#2) \expandafter+\usecentroid{#3#1} circle (1mm);
  \fi
}

\newcommand\monsterx[3][]{%
  \node [image,#1] at (#2) {\includegraphics{named/#3.png}};
  \draw [outline,#1] (#2) \usepath{#3};
  \edef\drawme{%
    \noexpand\draw
       [blue,fill,rotate around={\unangle[#1]:(#2)}] (#2)
          +\usecentroid{#3} circle (1mm);
      }
  \ifdots
    \drawme
  \fi
}

\newcommand\fullmonster[4]{% coordinate image path imageparms outlineparms
  \node [image,#4] at (#1) {\includegraphics{named/#2.png}};
  \draw [outline,#4] (#1) \usepath{#3};
  \edef\drawme{%
    \noexpand\draw
       [blue!50!black,fill,rotate around={\unangle[#4]:(#1)}] (#1)
          +\usecentroid{#3} circle (1mm);
      }
  \ifdots
    \drawme
  \fi
}

\newcommand\xfullmonster[5]{% coordinate image path imageparms outlineparms gfxparms
  \node [image,#4] at (#1) {\includegraphics[#5]{named/#2.png}};
  \draw [outline,#4] (#1) \usepath{#3};
  \edef\drawme{%
    \noexpand\draw
       [black,fill,rotate around={\unangle[#4]:(#1)}] (#1)
          +\usecentroid{#3} circle (1mm);
      }
  \ifdots
    \drawme
  \fi
}
